Petersonovo reseni ^WpLu64gB

#define N 2

int turn;
int interested[N];

void enterCS(int process) {
    int otherProcess = 1 - process;

    interested[process] = 1;
    turn = process;

    while ((turn == process) && (interested[otherProcess] == 1));
}

void leaveCS(int process) {
    interested[process] = 1;
}  ^hPCMb0JA
