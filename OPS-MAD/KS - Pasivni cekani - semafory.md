- Struktura semaforu

        int value;                    // "Hodnota" semaforu
        struct process *list;     // Fronta procesu stojicich "pred semaforem"
} semaphore;

- Operace nad semaforem jsou pak implementovany jako nedelitelne s touhle semantikou

    void wait(semaphore S) {
        S.value = S.value - 1;
        if (S.value < 0) {       // Je-li treba, zablokuj volajici proces a zarad ho
            block(S.list);       //  do fronty pred semaforem (S.list) 
    }
}

    void signal(semaphore S) {
        S.value = S.value + 1;
        if (S.value <= 0) {
            if (S.list != NULL) {     // Kdyz fronta neni prazdna vyjmi
                ...                      // vyjmi proces P z fronty
                wakeup(P);          // Probud proces P
        }   
    }
} ^OmpL0bjX

               volani wait bez blokovani)

    - signal() - Zvysuje hodnotu citace, pokud je nejaky proces ve fronte, je z ni vyjmut a odblokovan (probuzen) ^Pi7zsk9F
